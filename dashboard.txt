class PlayerStats {
  constructor(data) {
    this.id = data.id;
    this.team = data.team;
    this.role = data.role;
    this.stats = data.stats;
  }

  getStat(key) {
    return this.stats[key] ?? 0;
  }
}

let players = [];

async function loadMatchStats() {
  const res = await fetch('match.json');
  const data = await res.json();

  const extractPlayers = squad => [
    ...squad.onCourt,
    ...squad.bench
  ].map(p => new PlayerStats(p));

  players = [
    ...extractPlayers(data.squadA),
    ...extractPlayers(data.squadB)
  ];

  renderTable(players);
}

function sortByStat(statKey) {
  const sorted = [...players].sort(
    (a, b) => b.getStat(statKey) - a.getStat(statKey)
  );

  renderTable(sorted, statKey);
}


5ï¸âƒ£ Render tabella
function renderTable(list, statKey = null) {
  const tbody = document.getElementById('tableBody');
  tbody.innerHTML = '';

  list.forEach((p, index) => {
    const tr = document.createElement('tr');

    tr.innerHTML = `
      <td>${index + 1}</td>
      <td>${p.id}</td>
      <td>${p.team}</td>
      <td>${p.role ?? '-'}</td>
      <td>${statKey ? p.getStat(statKey) : ''}</td>
    `;

    tbody.appendChild(tr);
  });
}
